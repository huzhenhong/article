    # !usr/bin/env python
# -*- coding:utf-8 -*-

import json
import base64
import os
import tqdm
import cv2 as cv


def read_all_content(txt_path):
    img_content = {}
    file_name = None
    content_list = []
    is_first = True
    fr = open(txt_path)

    for line in fr.readlines():  # 逐行读取，滤除空格等
        if line.startswith("#"):
            if not is_first:
                img_content[file_name] = content_list.copy()
                content_list.clear()
            file_name = line[2:]
            is_first = False
        else:
            content_list.append(line)

    img_content[file_name] = content_list  # 最后一段
    return img_content

def generate_json(img_info, mark_info, is_landmark=False):
    global face_id
    face_id += 1
    print("process face count: ", face_id)

    im_path, im_h, im_w = img_info.strip().split(" ")

    str_json = {}
    shapes = []

    face_list = []
    landmark_list = []
    flags_list = []

    for line in mark_info:
        lineArr = line.strip().split(" ")

        face_list.append([
            [float(lineArr[0]), float(lineArr[1])],
            [float(lineArr[2]), float(lineArr[3])],
        ])

        if is_landmark is True:
            landmark_list.append([
                [[float(lineArr[4]), float(lineArr[5])]],
                [[float(lineArr[7]), float(lineArr[8])]],
                [[float(lineArr[10]), float(lineArr[11])]],
                [[float(lineArr[13]), float(lineArr[14])]],
                [[float(lineArr[16]), float(lineArr[17])]],
            ])

            flags_list.append([
                [float(lineArr[6])],
                [float(lineArr[9])],
                [float(lineArr[12])],
                [float(lineArr[15])],
                [float(lineArr[18])],
            ])

    for idx, face_info in enumerate(face_list):
        face_shape = {}
        face_shape["label"] = "face"
        face_shape["points"] = face_info
        face_shape["group_id"] = idx
        face_shape["shape_type"] = "rectangle"
        face_shape["flags"] = {}
        shapes.append(face_shape)

        if is_landmark is True:
            for landmark, flags in zip(landmark_list[idx], flags_list[idx]):
                landmark_shape = {}
                landmark_shape["label"] = "landmark"
                landmark_shape["points"] = landmark
                landmark_shape["group_id"] = idx
                landmark_shape["shape_type"] = "point"
                landmark_shape["flags"] = {}

                if flags[0] == -1:
                    landmark_shape["flags"]["-1"] = True
                elif flags[0] == 0:
                    landmark_shape["flags"]["0"] = True
                elif flags[0] == 1:
                    landmark_shape["flags"]["1"] = True
                else:
                    print("error: flags[0]: ", flags[0])

                shapes.append(landmark_shape)

    str_json["version"] = "5.0.1"
    str_json["flags"] = {"-1": False, "0": False, "1": False}
    str_json["shapes"] = shapes
    str_json["imagePath"] = os.path.join(img_root, im_path)
    im = cv.imread(str_json["imagePath"])
    str_json["imageHeight"] = im.shape[0]
    str_json["imageWidth"] = im.shape[1]
    str_json["imageData"] = base64encode_img(str_json["imagePath"])

    return str_json


def base64encode_img(image_path):
    with open(image_path, "rb") as f:
        imageData = f.read()
        imageData = base64.b64encode(imageData).decode("utf-8")
        return imageData

    # !usr/bin/env python
# -*- coding:utf-8 -*-

import os
import json
from tqdm import tqdm


if __name__ == "__main__":
    print(os.getcwd())

    dir_path_list = ["0-10", "11-20", "21-30", "31-40", "41-50", "51-61"]

    with open("scrfd_label.txt", "w") as f:
        for dir_path in dir_path_list:

            folder_dir_list = [p for p in os.listdir(dir_path) if os.path.isdir(os.path.join(dir_path, p))]

            for one_folder_dir in folder_dir_list:
                img_json_dir = os.path.join(dir_path, one_folder_dir)

                json_file_list = os.listdir(img_json_dir)

                for json_file in tqdm(json_file_list, "progress..."):
                    if json_file.startswith(".") or not json_file.endswith(".json"):
                        continue

                    json_path = os.path.join(img_json_dir, json_file)
                    if (json_file == "0_Parade_Parade_0_805.json"):
                        print(json_file)

                    label_dict = json.load(open(json_path, "r"))

                    label_info_list = list()

                    height = label_dict["imageHeight"]
                    width = label_dict["imageWidth"]
                    loc_info_list = label_dict["shapes"]

                    line = "# " + os.path.join(one_folder_dir, json_file).replace(".json", ".jpg") + " " + str(width) + " " + str(height)
                    label_info_list.append(line)

                    face_line = ""
                    for loc_info in loc_info_list:
                        obj_name = loc_info.get("label")

                        if obj_name == "face":
                            loc = loc_info.get("points")
                            x0, y0 = loc[0]  # 左上角点
                            x1, y1 = loc[1]  # 右下角点

                            face_line = format(x0, '.5f') + " " + format(y0, '.5f') + " " + format(x1, '.5f') + " " + format(y1, '.5f')
                            label_info_list.append(face_line)

                        elif (obj_name == "left_eye" or obj_name == "right_eye" or obj_name == "nose" or obj_name == "left_mouse" or obj_name == "right_mouse"):
                            loc = loc_info.get("points")
                            x, y = loc[0]

                            flags = loc_info.get("flags")
                            if flags["-1"]:
                                flags = -1
                            elif flags["0"]:
                                flags = 0
                            elif flags["1"]:
                                flags = 1

                            label_info_list[-1] += (" " + format(x, '.5f') + " " + format(y, '.5f') + " " + format(flags, '.5f'))

                    for line in label_info_list:
                        f.write(line)
                        f.write("\n")
