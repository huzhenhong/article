linux命令

系统

# 修改用户名，需要重启
sudo hostname new_name

# 查看CPU个数
cat /proc/cpuinfo | grep "physical id" | uniq | wc -l # uniq 删除重复行，wc –l 统计行数
# 查看CPU核数
cat /proc/cpuinfo | grep "cpu cores" | uniq
# 查看CPU型号
cat /proc/cpuinfo | grep 'model name' |uniq
# 查看内存总数
cat /proc/meminfo | grep MemTotal
# 查看硬盘大小
fdisk -l | grep Disk

# 查看内核版本
uname -a
cat /etc/redhat-release # 这种方法只适合Redhat系的Linux

# 更新源列表
sudo apt update

# 更新已安装的包
sudo apt upgrade

# 更新软件、升级系统
sudo apt dist-upgrade

# 手动更新系统
sudo apt full-upgrade

# 安装 C++ 构建套件，包括 gcc、g++ 和 make
sudo apt install build-essential

# 查看 CUDA 版本
nvcc --version

# 查看 cuDNN 版本
cat /usr/local/cuda/include/cudnn.h | grep CUDNN_MAJOR -A 2
cat /usr/local/cuda-10.2/targets/x86_64-linux/include/cudnn_version.h  | grep CUDNN_MAJOR -A 2

# 查看TensorRT版本
dpkg -l | grep TensorRT # 适用于ubuntu

# 添加环境变量
vim ~/.bashrc		# 在末尾添加 export PATH=$PATH:/need/add/path
source ~/.bashrc	# 立即生效

# 查找进程
# a：显示所有用户启动的进程
# u：以用户为主的格式来显示 
# x：显示当前用户启动的进程
ps -aux | grep xxx # USER PID %CPU %MEM VSZ RSS TTY STAT START TIME COMMAND

# 批量匹配 kill 进程
ps -ux | grep xxx | grep -v grep | awk '{print "kill -9 "$2}'

# 查看所有进程
ps -ef 

# kill 当前用户的所有进程
killall -u username

# 查看进程信息
top [-d number] | top [-bnp]
-d number	number代表秒数，表示top命令显示的页面更新一次的间隔 (default=5s)
-b	以批次的方式执行 top
-n	与-b配合使用，表示需要进行几次top命令的输出结果
-p	指定特定的 pid 进程号进行观察

top 命令显示的页面还可以输入以下按键执行相应的功能（注意大小写区分的）
参数	含义
？	显示在top当中可以输入的命令
P	以CPU的使用资源排序显示
M	以内存的使用资源排序显示
N	以pid排序显示
T	由进程使用的时间累计排序显示
k	给某一个pid一个信号,可以用来杀死进程(9)
r	给某个pid重新定制一个nice值（即优先级)
q	退出top（用ctrl+c也可以退出top）

文件

ln -s src link 		# 创建软连接
rm -rf  link 			# 创建软连接
ln -snf src link 	# 修改软连接

# 自适应单位显示文件大小
ls -lh

# 查看指定路径剩余空间
df -hl [dir]	# 全部
du -sh ./*		# 各个子文件夹大小
du -sh [dir]	# 查看当前目录总大小

# 查找文件
find / -name httpd.conf # 名字要完整，或者*匹配
locate	xxx				# 名字要完整，或者*匹配，可能需要提前 updatedb

# 文件拷贝
cp [-r] /src/path/ /dst/path/					# 简单拷贝
scp username@127.0.0.1:/src/path D:\dst\path	# 拷贝到windows
rsync -avuzP [--exclude={.*,*.a,*.b, c} user@x.x.x.x:/src/path/ /dst/path/	# linux之间拷贝

# 删除目录下的指定文件
find /root/ -type f -name "*.txt" | xargs rm
find /root/ -type f -name "*.txt" -exec rm {} \

# 统计输出信息的行数
wc -l

# 统计指定后缀文件个数
find dir '*.jpg' | wc -l

# 统计文件/文件夹数量
ls -l | grep "^-"| wc -l				# 查看当前目录下的文件数量（不包含子目录中的文件）
ls -lR | grep "^-"| wc -l				# 查看当前目录下的文件数量（包含子目录中的文件, R代表子目录）
ls -lR xxx*/ | grep "^-"| wc -l	# 查询当前路径下的指定前缀名的目录下的所有文件数量
ls -l | grep "^d"| wc -l				# 查看当前目录下的文件夹目录个数（不包含子目录中的目录）
ls -lR | grep "^d"| wc -l				# 查看当前目录下的文件夹目录个数（包含子目录中的目录, R代表子目录）
ls -lR xxx*/ | grep "^d"| wc -l	# 查询当前路径下的指定前缀名的目录下的所有目录数量

# 统计目录文件数量脚本，bash count.sh xxx，输出总数及各个子目录中文件数
#!/bin/sh
numOfArgs=$#
if [ $numOfArgs -ne 1 ]; then
echo -e "Usage: \nbash $0 dirForCount"
exit -1
fi
# args
ROOTDIR=$1
# core part
find $ROOTDIR -maxdepth 1 -type d | sort | while read dir; do
count=$(find "$dir" -type f | wc -l)
echo "$dir: $count"
done

# 反向删除
rm -rf !(FileName)

其它

# 返回上次目录
cd -

# 执行命令
your-command&				# 后台执行
nohup your-command  # 不挂断执行，追加 [>what-you-want.log 2>&1] 输出重定向到文件
jobs  [-l] 					# 查看当前终端后台运行的任务
kill %jobID  				# 结束任务
fg %jobID  					# 调至前台运行
Crtl + z 						# 前台任务暂停至后台
bg %jobID  					# 后台暂停任务继续执行

# 执行脚本
. /path/to/script.sh			# 在当前 shell 中执行
source /path/to/script.sh	# 在当前 shell 中执行

磁盘

sudo -s							# 进入Single User模式
mkdir /Volumes/EFI	# 在Volumes下创建EFI文件夹
diskutil list				# 一般的EFI分区会在disk0s1
mount -t msdos /dev/disk0s1 /Volumes/EFI	# 挂载分区
diskutil unmount /dev/disk0s1 						# 卸载分区

软件源

config-manager --add-repo https://xx.repo # 添加源
rm -f ${文件名}.repo												# 删除源
