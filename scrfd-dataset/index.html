scrfd 数据集

  • 标签格式

    # 图片名称 图片宽 图片高
x1, y1, x2, y2, kps_x1, kps_y1, flag1, kps_x2, kps_y2, flag2, kps_x3, kps_y3, flag3, kps_x4, kps_y4, flag4, kps_x5, kps_y5, flag5 
x1, y1, x2, y2, kps_x1, kps_y1, flag1, kps_x2, kps_y2, flag2, kps_x3, kps_y3, flag3, kps_x4, kps_y4, flag4, kps_x5, kps_y5, flag5

    landmarks的flags说明

    • -1：没有关键点，此时关键点的坐标也全为-1.00000

    • 0：有关键点，人脸清晰

    • 1：有关键点，人脸模糊或侧脸

​ 特别注意，在scrfd源代码里面读取标签时不能有空行存在

  • 根据widerface标志清洗后得到的有效人脸，进一步清洗scrfd标签文件，代码如下

    # !usr/bin/env python
# -*- coding:utf-8 -*-

import os
import cv2 as cv
from tqdm import tqdm


def read_scrfd(filename):
    sub_content_map = {}

    with open(filename) as f:
        content = f.readlines()
        content = [line.strip() for line in content]

        content.append('end')
        for line in content:
            if line == 'end':
                break
            elif line.startswith('#'):
                # im_name, width, height = line[2:].split()
                im_name = line
                sub_content_map[im_name] = []
            else:
                sub_content_map[im_name].append(line)

    return sub_content_map


def read_wider_face(filename):
    sub_content_map = {}

    with open(filename) as f:
        content = f.readlines()
        content = [line.strip() for line in content]

        content.append('end')
        for line in content:
            if line == 'end':
                break
            elif line.endswith('.jpg'):
                im_name = line
                # im_name, width, height = line[2:].split()
                sub_content_map[im_name] = []
            else:
                sub_content_map[im_name].append(line)

    return sub_content_map


def is_valid_face(key, val, content):
    if key not in content.keys():
        return False

    val = [int(float(i)) for i in val.split()]

    sub_content = content[key]
    for line in sub_content:
        line = [int(i) for i in line.split()]
        if line[0] == val[0] and line[1] == val[1]:
            return True
    return False


def save_content(filename, centent):
    with open(filename, 'w') as f:
        for k, v in centent.items():
            if len(v) == 0:
                continue

            f.write(k)
            f.write('\n')

            for line in v:
                f.write(line)
                f.write('\n')

    invalid_cnt = 0

    for k, v in tqdm(scrfd_content.items(), '...'):
        valid_content[k] = []
        m_name, width, height = k[2:].split()
        for line in v:
            ret = is_valid_face(m_name, line, valid_wider_face_content)
            if ret:
                valid_content[k].append(line)
            else:
                invalid_cnt += 1
            total_cnt += 1

    print('valid_cnt: ', total_cnt - invalid_cnt)
    print('invalid_cnt: ', invalid_cnt)
    print('total_cnt: ', total_cnt)
    
    save_content(clean_label_path, valid_content)

从scrfd数据里制作大头贴数据集

  • 这里简单的通过人脸面积来筛选，其实还应该加上扣取的大头贴里是否还包含其它人脸

  • 对人脸的patch大小比较值得推敲，原先希望将四个方向的patch设为宽高的0~0.5倍数，实际测试发现不行，基本上都是抠太多了，也就是系数还是太大，未再进一步测试，沿用原有的在四个方向均随机扣取0～50个像素

  • 数据集制作代码

    # !usr/bin/env python
# -*- coding:utf-8 -*-
import os
import shutil
import cv2 as cv
from tqdm import tqdm
import random
import numpy as np


def read_scrfd(filename):
    sub_content_map = {}

    with open(filename) as f:
        content = f.readlines()
        content = [line.strip() for line in content]

        content.append('end')
        for line in content:
            if line == 'end':
                break
            elif line.startswith('#'):
                # im_name, width, height = line[2:].split()
                im_name = line
                sub_content_map[im_name] = []
            else:
                sub_content_map[im_name].append(line)

    return sub_content_map


def split_content(content):
    totoal_line = len(content)
    sub_content_map = {}
    idx = 0
    while idx < totoal_line:
        im_name = content[idx]
        bbox_num = int(content[idx + 1])
        if bbox_num == 0:
            bbox_num = 1

        bbox_info = content[idx + 2:idx + 2 + bbox_num]
        idx = idx + 2 + bbox_num
        sub_content_map[im_name] = bbox_info
    print(idx)
    return sub_content_map


def read_label_info(labelfile):
    with open(labelfile) as f:
        content = f.readlines()
        content = [line.strip() for line in content]
        return split_content(content)


def compute_iou(rec1, rec2):
    """
    computing IoU
    :param rec1: (y0, x0, y1, x1), which reflects
            (top, left, bottom, right)
    :param rec2: (y0, x0, y1, x1)
    :return: scala value of IoU
    """
    # computing area of each rectangles
    S_rec1 = (rec1[2] - rec1[0]) * (rec1[3] - rec1[1])
    S_rec2 = (rec2[2] - rec2[0]) * (rec2[3] - rec2[1])

    # computing the sum_area
    sum_area = S_rec1 + S_rec2

    # find the each edge of intersect rectangle
    left_line = max(rec1[1], rec2[1])
    right_line = min(rec1[3], rec2[3])
    top_line = max(rec1[0], rec2[0])
    bottom_line = min(rec1[2], rec2[2])

    # judge if there is an intersect
    if left_line >= right_line or top_line >= bottom_line:
        return 0
    else:
        intersect = (right_line - left_line) * (bottom_line - top_line)
        return (intersect / (sum_area - intersect)) * 1.0


def is_only_one_face(face, patch, face_list):
    for line in face_list:
        line = [int(float(i)) for i in line.split()]

        if line[0:4] == face[0:4]:
            continue

        if (compute_iou(line[0:4], patch[0:4])) > 0:
            return False

    return True


def save_label_to_txt(sava_file, content):
    with open(sava_file, 'w') as f:
        for k, v in content.items():
            f.write(k)
            f.write('\n')
            f.write(v)
            f.write('\n')


if __name__ == '__main__':
    print('run path: ', os.getcwd())

    # img_dir = '/Users/huzh/Documents/dataset/face/wider-face/original'
    img_dir = '/Users/huzh/Documents/dataset/face/wider-face/gaoyun-clean-train-2023-02-01'

    is_train = True
    # is_train = False
    if is_train:
        label_path = 'file/train-clean-2023-02-01.txt'
        sava_file = 'file/train-clean-2023-02-01-bigface.txt'
        save_dir = 'file/DTT_train'
    else:
        label_path = 'file/val-clean-2023-02-01.txt'
        sava_file = 'file/val-clean-2023-02-01-bigface.txt'
        save_dir = 'file/DTT_val'

    if os.path.exists(save_dir):
        shutil.rmtree(save_dir)
    os.makedirs(save_dir)

    scrfd_content = read_scrfd(label_path)

    sigal_face = 0
    total_cnt = 0
    valid_cnt = 0
    valid_content = {}

    for k, v in tqdm(scrfd_content.items(), '...'):
        if len(v) < 2:
            sigal_face += 1

        if len(v) == 0:
            continue

        idx = 0

        # 挑选最大人脸
        max_face_line = 0
        max_face_area = 0

        for line in v:
            idx += 1
            line = [int(float(i)) for i in line.split()]

            face_area = (line[2] - line[0]) * (line[3] - line[1])
            if face_area > max_face_area:
                max_face_area = face_area
                max_face_line = line

        # 过滤小面积人脸
        if max_face_area < 5000:
            continue

        # 坐标为负
        if max_face_line[0] < 0 or max_face_line[1] < 0 or max_face_line[2] < 0 or max_face_line[
                3] < 0:
            continue

        dir_name, im_name = k.split('/')
        dir_name = dir_name.split()[1]
        im_name = im_name.split()[0]
        print('process image: ', im_name)

        if is_train:
            im = cv.imread(os.path.join(img_dir, dir_name, im_name))
        else:
            im = cv.imread(os.path.join(img_dir, dir_name, im_name))

        height, width, _ = im.shape

        # 坐标超过宽高
        if max_face_line[0] > width or max_face_line[1] > height or max_face_line[
                2] > width or max_face_line[3] > height:
            continue

        # cut face
        patch = np.array(max_face_line[0:4])

        if random.randint(0, 10) < 8:
            expand = 50
            left_expand = random.randint(-expand, 0)
            height_expand = random.randint(-expand, 0)
            right_expand = random.randint(0, expand)
            bottom_expand = random.randint(0, expand)
            patch = patch + (left_expand, height_expand, right_expand, bottom_expand)

            patch[:2] = patch[:2].clip(min=0)
            patch[2:3] = patch[2:3].clip(max=width)
            patch[3:4] = patch[3:4].clip(max=height)

            # 实际的expand
            left_expand = max_face_line[0] - patch[0]
            height_expand = max_face_line[1] - patch[1]
        else:
            left_expand = 0
            height_expand = 0

        # patch 里只包含一个人脸
        if not is_only_one_face(max_face_line, patch, v):
            continue

        valid_cnt += 1

        if is_train:
            bbox_info = [
                left_expand,
                height_expand,
                max_face_line[2] - max_face_line[0] + left_expand,
                max_face_line[3] - max_face_line[1] + height_expand,
                max_face_line[4] - max_face_line[0] +
                left_expand if max_face_line[4] >= 0 else -1.0,
                max_face_line[5] - max_face_line[1] +
                height_expand if max_face_line[4] >= 0 else -1.0,
                max_face_line[6],
                max_face_line[7] - max_face_line[0] +
                left_expand if max_face_line[4] >= 0 else -1.0,
                max_face_line[8] - max_face_line[1] +
                height_expand if max_face_line[4] >= 0 else -1.0,
                max_face_line[9],
                max_face_line[10] - max_face_line[0] +
                left_expand if max_face_line[4] >= 0 else -1.0,
                max_face_line[11] - max_face_line[1] +
                height_expand if max_face_line[4] >= 0 else -1.0,
                max_face_line[12],
                max_face_line[13] - max_face_line[0] +
                left_expand if max_face_line[4] >= 0 else -1.0,
                max_face_line[14] - max_face_line[1] +
                height_expand if max_face_line[4] >= 0 else -1.0,
                max_face_line[15],
                max_face_line[16] - max_face_line[0] +
                left_expand if max_face_line[4] >= 0 else -1.0,
                max_face_line[17] - max_face_line[1] +
                height_expand if max_face_line[4] >= 0 else -1.0,
                max_face_line[18],
            ]
        else:
            bbox_info = [
                left_expand,
                height_expand,
                max_face_line[2] - max_face_line[0] + left_expand,
                max_face_line[3] - max_face_line[1] + height_expand,
            ]

        face = im[patch[1]:patch[3], patch[0]:patch[2]]

        # for debug
        # face = cv.rectangle(face, (bbox_info[0], bbox_info[1]),
        #                     (bbox_info[2], bbox_info[3]), (0, 0, 255), 2)

        # if bbox_info[4] >= 0:
        #     face = cv.circle(face, (bbox_info[4], bbox_info[5]), 2, (0, 255, 0), -1)
        #     face = cv.circle(face, (bbox_info[7], bbox_info[8]), 2, (0, 255, 0), -1)
        #     face = cv.circle(face, (bbox_info[10], bbox_info[11]), 2, (0, 255, 0), -1)
        #     face = cv.circle(face, (bbox_info[13], bbox_info[14]), 2, (0, 255, 0), -1)
        #     face = cv.circle(face, (bbox_info[16], bbox_info[17]), 2, (0, 255, 0), -1)

        save_name = os.path.join(save_dir, f'{im_name[:-4]}-{idx}.jpg')
        cv.imwrite(save_name, face)

        bbox_info = [f'{i:.5f}' for i in bbox_info]
        bbox_info_str = " ".join(bbox_info)
        h, w, _ = face.shape
        bigface_name = f'# {save_name} {w} {h}'
        valid_content[bigface_name] = bbox_info_str

        total_cnt += 1

    print(valid_cnt, total_cnt)

    save_label_to_txt(sava_file, valid_content)
